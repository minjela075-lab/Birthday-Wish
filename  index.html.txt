<script>
const text="Happy 19th Birthday Angel 🎂❤️";
let i=0;
function type(){
  if(i<text.length){
    document.getElementById("typing").innerHTML+=text[i];
    i++;
    setTimeout(type,50);
  }
}
type();

function startReveal(){
  document.getElementById("reveal-screen").classList.add("fade-out");
  document.getElementById("music").play();
}

function blowCandles(){
  document.querySelectorAll(".flame").forEach(f=>f.classList.add("off"));
  document.getElementById("wish").innerText="Wish made ❤️";
}

function gift(){
  document.getElementById("giftText").innerText="Your gift is my love ❤️";
}

function openLetter(){
  document.getElementById("popup").classList.add("show");
  document.getElementById("letter").innerText=
`Happy 19th Birthday Angel ❤️

I don’t even know where to start, but I just want you to know how special you are to me.

Even though we’re far apart, you’re always on my mind. You’re someone who makes my days better just by existing, and I’m really grateful to have you in my life.

Your smile, your energy, the way you talk, everything about you just feels different… in a good way. You make things feel lighter, happier, and more meaningful.

I wish I could be there with you today, celebrate with you, see you smile in person, and make your day even more special. But even from far away, I hope you can feel how much I care about you.

You deserve all the happiness, love, and success in the world, not just today but every day.

Stay the same beautiful, amazing person you are… because that’s exactly why you mean so much to me.

Happy birthday again, Angel ❤️🎂`;
}

function closeLetter(){
  document.getElementById("popup").classList.remove("show");
}
